      return (
        <Sequence key={`title-${i}`} from={fromFrame} durationInFrames={durationInFrames}>
          <TitleOverlay
            text={title.text}
            subtitle={title.subtitle}
            style={title.style}
            durationMs={title.durationMs}
          />
        </Sequence>
      );
