From CONTEXT.md decisions:
- D-01: Single shared Docker named volume `pipeline-data`, mounted at `/data/pipeline` in all containers
- D-02: Job-scoped subdirectories: `/data/pipeline/{job_id}/`
- D-03: Containers receive metadata via env vars only: PIPELINE_JOB_ID, INPUT_PATH, OUTPUT_PATH
- D-04: Direct read-write mount pattern. Input dir read-only, output dir read-write
- D-10: Job-scoped directory pattern with step-name subdirectories: `/data/pipeline/{job_id}/{step_name}/`
- D-11: Input MP4 lives in job root: `/data/pipeline/{job_id}/input/video.mp4`
- D-12: Fixed predictable filenames (transcript.json, silence-cuts.json, output.mp4, manifest.json)
- D-13: Standardized output filenames across all steps
</interfaces>

1. Create `shared/constants.ts` with exported path constants and conventions:

   ```typescript
   // Shared pipeline constants — all containers must use these paths

   /** Base directory where the shared Docker volume is mounted in containers */
   export const PIPELINE_DATA_DIR = "/data/pipeline";

   /** Directory for input video file within a job (per D-11) */
   export const INPUT_DIR_NAME = "input";

   /** Filename for the source video (per D-12) */
   export const INPUT_VIDEO_FILENAME = "video.mp4";

   /** Full input path template: /data/pipeline/{jobId}/input/video.mp4 */
   export function inputPath(jobId: string): string {
     return `${PIPELINE_DATA_DIR}/${jobId}/${INPUT_DIR_NAME}/${INPUT_VIDEO_FILENAME}`;
   }

   /** Directory for a step's output within a job (per D-10) */
   export function stepOutputDir(jobId: string, stepName: string): string {
     return `${PIPELINE_DATA_DIR}/${jobId}/${stepName}`;
   }

   /** Standardized output filenames (per D-13) */
   export const OUTPUT_FILENAMES = {
     transcript: "transcript.json",
     silenceCuts: "silence-cuts.json",
     outputVideo: "output.mp4",
     manifest: "manifest.json",
   } as const;

   /** Manifest output path for a step (per D-07) */
   export function manifestPath(jobId: string, stepName: string): string {
     return `${stepOutputDir(jobId, stepName)}/${OUTPUT_FILENAMES.manifest}`;
   }

   /** Known pipeline step names (used for directory naming) */
   export const STEP_NAMES = {
     whisper: "whisper",
     silenceCutter: "silence-cutter",
     remotionRenderer: "remotion-renderer",
     ffmpegFinalizer: "ffmpeg-finalizer",
   } as const;

   /** Exit codes (per D-06): 0=success, 1=generic error, 2+=step-specific */
   export const EXIT_CODES = {
     SUCCESS: 0,
     GENERIC_ERROR: 1,
     // Step-specific codes start at 2 — each container defines its own
   } as const;

   /** Directory structure template for a new job */
   export function jobDirStructure(jobId: string): Record<string, string> {
     return {
       jobRoot: `${PIPELINE_DATA_DIR}/${jobId}`,
       inputDir: inputPath(jobId).replace(`/${INPUT_VIDEO_FILENAME}`, ""),
       inputFile: inputPath(jobId),
     };
   }
   ```
