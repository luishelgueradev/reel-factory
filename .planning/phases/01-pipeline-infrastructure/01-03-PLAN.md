1. Create `services/base-python/Dockerfile`:
   ```dockerfile
   # Base Python image for pipeline step containers
   # All Python-based steps (Whisper, Silence Cutter, FFmpeg Finalizer) inherit from this image.
   # FFmpeg is pinned to ensure version consistency across all pipeline containers (PIPE-05, D-06 in PITFALLS).

   ARG FFMPEG_VERSION=7.1.1

   FROM python:3.12-slim AS base

   ARG FFMPEG_VERSION

   # Install FFmpeg static build from johnvansickle (official static builds)
   # This ensures the exact same FFmpeg version across Python and Node containers
   RUN apt-get update && apt-get install -y --no-install-recommends \
       curl \
       xz-utils \
       ca-certificates \
     && curl -L "https://johnvansickle.com/ffmpeg/releases/ffmpeg-release-amd64-static.tar.xz" -o /tmp/ffmpeg.tar.xz \
     && tar -xf /tmp/ffmpeg.tar.xz -C /tmp \
     && cp /tmp/ffmpeg-${FFMPEG_VERSION}-amd64-static/ffmpeg /usr/local/bin/ \
     && cp /tmp/ffmpeg-${FFMPEG_VERSION}-amd64-static/ffprobe /usr/local/bin/ \
     && chmod +x /usr/local/bin/ffmpeg /usr/local/bin/ffprobe \
     && rm -rf /tmp/ffmpeg* \
     && apt-get purge -y curl xz-utils \
     && apt-get autoremove -y \
     && apt-get clean \
     && rm -rf /var/lib/apt/lists/*

   # Verify FFmpeg installation and version
   RUN ffmpeg -version | head -1 && ffprobe -version | head -1

   # Create pipeline data directory (matches shared/constants.ts PIPELINE_DATA_DIR)
   RUN mkdir -p /data/pipeline

   WORKDIR /app

   # Default: run the step's main entry point
   # Individual step containers override CMD with their specific entry point
   CMD ["python", "main.py"]
   ```

2. Create `services/base-node/Dockerfile`:
   ```dockerfile
   # Base Node.js image for pipeline step containers
   # All Node-based steps (Remotion Renderer, Orchestrator) inherit from this image.
   # Uses Debian bookworm-slim (NOT Alpine — per Remotion docs and PITFALLS.md Pitfall 7).

   ARG FFMPEG_VERSION=7.1.1

   FROM node:22-bookworm-slim AS base

   ARG FFMPEG_VERSION

   # Install FFmpeg static build (same version as Python container for PIPE-05)
   # Also install Chrome dependencies required by Remotion (PITFALLS.md Pitfall 3)
   RUN apt-get update && apt-get install -y --no-install-recommends \
       curl \
       xz-utils \
       ca-certificates \
       # Chrome Headless Shell dependencies (PITFALLS.md Pitfall 3, Remotion Docker docs)
       libnss3 \
       libdbus-1-3 \
       libatk1.0-0 \
       libgbm-dev \
       libasound2 \
       libxrandr2 \
       libxkbcommon-dev \
       libxfixes3 \
       libxcomposite1 \
       libxdamage1 \
       libatk-bridge2.0-0 \
       libpango-1.0-0 \
       libcairo2 \
       libcups2 \
     && curl -L "https://johnvansickle.com/ffmpeg/releases/ffmpeg-release-amd64-static.tar.xz" -o /tmp/ffmpeg.tar.xz \
     && tar -xf /tmp/ffmpeg.tar.xz -C /tmp \
     && cp /tmp/ffmpeg-${FFMPEG_VERSION}-amd64-static/ffmpeg /usr/local/bin/ \
     && cp /tmp/ffmpeg-${FFMPEG_VERSION}-amd64-static/ffprobe /usr/local/bin/ \
     && chmod +x /usr/local/bin/ffmpeg /usr/local/bin/ffprobe \
     && rm -rf /tmp/ffmpeg* \
     && apt-get purge -y curl xz-utils \
     && apt-get autoremove -y \
     && apt-get clean \
     && rm -rf /var/lib/apt/lists/*

   # Verify FFmpeg installation and version
   RUN ffmpeg -version | head -1 && ffprobe -version | head -1

   # Create pipeline data directory (matches shared/constants.ts PIPELINE_DATA_DIR)
   RUN mkdir -p /data/pipeline

   WORKDIR /app

   # Default: run the step's main entry point
   CMD ["node", "dist/index.js"]
   ```

   Note: The Chrome dependencies are installed in the base Node image so that the Remotion container (Phase 5) doesn't need additional OS packages. For non-Remotion Node containers, these are unused but harmless (~30MB overhead).
